### 快速排序

```cpp
 void quick_sort(int q[], int l, int r)          //基于霍尔分区
{
    if (l >= r) return;
    int i = l - 1, j = r + 1, x = q[l + r >> 1];
    while (i < j)
    {
        do i ++ ; while (q[i] < x);
        do j -- ; while (q[j] > x);   //do...while... 避免了两边遇到枢轴量陷入死循环
        if (i < j) swap(q[i], q[j]);
    }
    quick_sort(q, l, j), quick_sort(q, j + 1, r);
}
```

> 空间复杂度         //取决于递归栈的深度
>
> 最好：O($log_2$n)	最坏：O(n)	平均：O($log_2$n)
>
> 时间复杂度         //取决于原数据的对称性
>
> 最好：O(n$log_2$n)	最坏：O($n^2$)	平均：O(n$log_2$n)

```cpp
void merge_sort(int q[], int l, int r)
{
    if (l >= r) return;
    int mid = l + r >> 1;
    merge_sort(q, l, mid);
    merge_sort(q, mid + 1, r);
    int k = 0, i = l, j = mid + 1;
    while (i <= mid && j <= r)
        if (q[i] <= q[j]) tmp[k ++ ] = q[i ++ ];
        else tmp[k ++ ] = q[j ++ ];
    while (i <= mid) tmp[k ++ ] = q[i ++ ];
    while (j <= r) tmp[k ++ ] = q[j ++ ];

}
```

>

### 整数二分

```cpp
bool check(int x) {/* ... */} // 检查x是否满足某种性质

// 查找左区间的右边界
int bsearch_1(int l, int r)
{
    while (l < r)
    {
        int mid = l + r + 1 >> 1;   //l+r+1避免陷入死循环
        if (check(mid)) l = mid;
        else r = mid - 1;
    }
    return l;
}

// 查找右区间的左边界
int bsearch_2(int l, int r)
{
    while (l < r)
    {
        int mid = l + r >> 1;
        if (check(mid)) r = mid;    // check()判断mid是否满足性质
        else l = mid + 1;
    }
    return l;
}
```

### 一维前缀和

```cpp
S[i] = a[1] + a[2] + ... a[i]
a[l] + ... + a[r] = S[r] - S[l - 1]
```

> 用处：快速求一段区间的和：$a[i]+a[i+1]+......+a[j] = S[j]-S[i-1]$
>
> 一维前缀和数组构建方法：$S[0]=0$ ,$S[i]=S[i-1]+a[i]$



***

### 二维前缀和

```cpp
S[i, j] = 第i行j列格子左上部分所有元素的和
以(x1, y1)为左上角，(x2, y2)为右下角的子矩阵的和为：
S[x2, y2] - S[x1 - 1, y2] - S[x2, y1 - 1] + S[x1 - 1, y1 - 1]
```

> 用处：快速求一块区域的和：$S[x2,y2]-S[x1-1,y2]-S[x2,y1-1]+S[x1-1,y1-1]$
>
> 二维前缀和数组构建方法：$S[i,j]=S[i-1,j]+S[i,j-1]-S[i-1,j-1]+a[i,j]$

### 一维差分

```cpp
给区间[l, r]中的每个数加上c：B[l] += c, B[r + 1] -= c
```

> 用处：给一段区间的数统一加一个常数c：先对差分数组进行$O(1)$的操作，然后由差分数组还原得到原数组
>
> 一维差分数组的构建：首先假设原数组与差分数组所有元素为0，此时满足差分数组性质；然后一个一个插入元素还原数组，得到差分数组，插入公式为$B[i]+=A[i],B[i+1]-=A[i]$



***

### 二维差分

```cpp
给以(x1, y1)为左上角，(x2, y2)为右下角的子矩阵中的所有元素加上c：
S[x1, y1] += c, S[x2 + 1, y1] -= c, S[x1, y2 + 1] -= c, S[x2 + 1, y2 + 1] += c
```

> 用处：给一块区域的数统一加一个常数c：先对先对差分数组进行$O(1)$的操作，然后由差分数组还原得到原数组
>
> 二维差分数组构建：首先假设原数组与差分数组所有元素为0，此时满足差分数组性质；然后一个一个插入元素还原数组，得到差分数组，插入公式为$B[i,j]+=A[i,j],B[i,j+1]-=A[i,j],B[i+1,j]-=A[i,j],B[i+1,j+1]=A[i,j]$

### 离散化

```cpp
vector<int> alls; // 存储所有待离散化的值
sort(alls.begin(), alls.end()); // 将所有值排序
alls.erase(unique(alls.begin(), alls.end()), alls.end());   // 去掉重复元素
// 二分求出x对应的离散化的值
int find(int x) // 找到第一个大于等于x的位置
{
    int l = 0, r = alls.size() - 1;
    while (l < r)
    {
        int mid = l + r >> 1;
        if (alls[mid] >= x) r = mid;
        else l = mid + 1;
    }
    return r + 1; // 映射到1, 2, ...n
}
```

> 适用于当数据个数很少，但是涵盖范围很大时，进行离散化节约空间
>
> 离散化的核心在于保留数据间的大小关系，而不在乎数据实际数值
>
> **注意**：所以涉及到的点都需要离散化，包括初始数据和后续操作数据

### 合并区间

```cpp
typedef pair<int,int> PII;

void merge(vector<PII>& segs)
{
    vector<PII> res;
    int st=-2e9  ,  ed=-2e9;
    for(auto seg : segs)
    {
        if(seg.first>ed)
        {
            if(st!=-2e9)  res.push_back({st,ed});
            st=seg.first , ed=seg.second;
        }
        else
        {
            ed=max(ed,seg.second);
        }
    }
    if(st!=-2e9)  res.push_back({st,end});
    segs=res;
}
```

### 高精度运算

````cpp
// C = A + B, A >= 0, B >= 0
vector<int> add(vector<int> &A, vector<int> &B)
{
    if (A.size() < B.size()) return add(B, A);

    vector<int> C;
    int t = 0;
    for (int i = 0; i < A.size(); i ++ )
    {
        t += A[i];
        if (i < B.size()) t += B[i];
        C.push_back(t % 10);
        t /= 10;
    }

    if (t) C.push_back(t);
    return C;
}

// C = A - B, 满足A >= B, A >= 0, B >= 0
vector<int> sub(vector<int> &A, vector<int> &B)
{
    vector<int> C;
    for (int i = 0, t = 0; i < A.size(); i ++ )
    {
        t = A[i] - t;
        if (i < B.size()) t -= B[i];
        C.push_back((t + 10) % 10);
        if (t < 0) t = 1;
        else t = 0;
    }

    while (C.size() > 1 && C.back() == 0) C.pop_back();
    return C;
}

// C = A * b, A >= 0, b >= 0
vector<int> mul(vector<int> &A, int b)
{
    vector<int> C;

    int t = 0;
    for (int i = 0; i < A.size() || t; i ++ )
    {
        if (i < A.size()) t += A[i] * b;
        C.push_back(t % 10);
        t /= 10;
    }
    while (C.size() > 1 && C.back() == 0) C.pop_back();
    return C;
}

// A / b = C ... r, A >= 0, b > 0
vector<int> div(vector<int> &A, int b, int &r)
{
    vector<int> C;
    r = 0;
    for (int i = A.size() - 1; i >= 0; i -- )
    {
        r = r * 10 + A[i];
        C.push_back(r / b);
        r %= b;
    }
    reverse(C.begin(), C.end());
    while (C.size() > 1 && C.back() == 0) C.pop_back();
    return C;
}
````
